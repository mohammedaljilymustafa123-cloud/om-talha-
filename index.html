-- 1. جدول المنتجات CREATE TABLE products ( id UUID PRIMARY KEY DEFAULT gen_random_uuid(), name_ar VARCHAR(255) NOT NULL, name_local VARCHAR(255) NOT NULL, description TEXT, category VARCHAR(100) NOT NULL, price_sdg DECIMAL(12, 2) NOT NULL, stock_quantity INTEGER NOT NULL DEFAULT 0, images_urls JSONB DEFAULT '[]'::jsonb, is_active BOOLEAN DEFAULT true, created_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP ); -- 2. جدول الطلبات CREATE TABLE orders ( id UUID PRIMARY KEY DEFAULT gen_random_uuid(), customer_name VARCHAR(255) NOT NULL, phone_number VARCHAR(50) NOT NULL, state VARCHAR(100) NOT NULL DEFAULT 'الجزيرة', city VARCHAR(100) NOT NULL DEFAULT 'ود مدني', address_details TEXT NOT NULL, payment_method VARCHAR(50) NOT NULL, -- 'cash', 'bankak', 'fawry' payment_receipt_url TEXT, -- رابط صورة الإشعار في حال التحويل transaction_id VARCHAR(100), -- الرقم المرجعي للتحويل delivery_fee DECIMAL(10, 2) NOT NULL DEFAULT 0.00, total_amount DECIMAL(12, 2) NOT NULL, order_status VARCHAR(50) DEFAULT 'pending', -- 'pending', 'confirmed', 'delivered', 'cancelled' created_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP ); -- 3. جدول عناصر الطلب (الربط بين المنتجات والطلبات) CREATE TABLE order_items ( id UUID PRIMARY KEY DEFAULT gen_random_uuid(), order_id UUID REFERENCES orders(id) ON DELETE CASCADE, product_id UUID REFERENCES products(id), quantity INTEGER NOT NULL DEFAULT 1, price_at_purchase DECIMAL(12, 2) NOT NULL );
